package npw;

import java.awt.Color;
import javax.swing.SwingUtilities;
import painter.SPainter;
import shapes.SCircle;
import shapes.SSquare;

public class funsquares{
    private void paintTheImage(){
        SPainter painter = new SPainter ( " Fun Squares " , 600 , 600) ;
        SSquare square = new SSquare (150) ;
        paintYellowSquare ( painter , square ) ;
        paintGraySquares ( painter , square ) ;
        paintBlueSquares ( painter , square );
        paintRedSquares ( painter , square );
    }

    private void paintRedSquares(SPainter klee, SSquare square) {
        klee.setColor(Color.RED);
        klee.mfd(150);
        klee.mrt(150);
        klee.paint(square);
        klee.mlt(150);
        klee.mbk(150);
        klee.mlt(150);
        klee.mfd(150);
        klee.paint(square);

    }

    private void paintBlueSquares(SPainter klee, SSquare square) {
        klee.setColor(Color.BLUE);
        square.s2();
        klee.mbk(5);
        klee.mrt(150);
        klee.paint(square);
        klee.mfd(5);
        klee.mlt(150);
        klee.mbk(5);
        klee.mlt(150);
        klee.paint(square);
        klee.mrt(150);
        klee.mfd(150);

    }

    private void paintGraySquares(SPainter klee, SSquare square) {
        klee.setColor(Color.GRAY);
        square.x2();
        klee.mfd(150);
        klee.paint(square);
        klee.mbk(150);
        klee.mrt(150);
        klee.paint(square);
        klee.mlt(300);
        klee.paint(square);
        klee.mrt(150);
        klee.mbk(150);
        klee.paint(square);

    }

    private void paintYellowSquare(SPainter klee, SSquare square) {
        square.s2();
        klee.setColor(Color.YELLOW);
        klee.paint(square);
    }

    public funsquares(){
        paintTheImage();
    }

    public static void main(String[] args){
        SwingUtilities.invokeLater(new Runnable(){
            public void run() {
                new funsquares();
            }
        });
    }
}
